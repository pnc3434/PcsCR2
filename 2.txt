using System;

class Program
{
    static void Main()
    {
        Console.Write("Введите 6-и значныйт номер билета: ");
        int ticket = int.Parse(Console.ReadLine());

        int digit1 = ticket / 100000;// Разбиваем число на цифры
        int digit2 = (ticket / 10000) % 10;
        int digit3 = (ticket / 1000) % 10;
        int digit4 = (ticket / 100) % 10;
        int digit5 = (ticket / 10) % 10;
        int digit6 = ticket % 10;
        int sumFirstThree = digit1 + digit2 + digit3;// Суммы первых трех и последних трех цифр
        int sumLastThree = digit4 + digit5 + digit6;

        if (sumFirstThree == sumLastThree)// Проверяем, является ли билет счастливым
        {
            Console.WriteLine("Счастливый билет");
        }
        else
        {
            Console.WriteLine("Обычный билет");
        }
    }
}
