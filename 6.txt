using System;

class Program
{
    static void Main()
    {
        Console.Write("Введите количество бактерий (N): ");
        int b = int.Parse(Console.ReadLine());

        Console.Write("Введите количество капель антибиотика (X): ");
        int x = int.Parse(Console.ReadLine());

        int h = 0;
        int k = x * 10;

        Console.WriteLine("\nДинамика изменения количества бактерий:");

        while (k > 0 && b > 0)
        {
            h++;

            // Удваиваем бактерии
            b = b * 2;

            // Действие антибиотика
            b = b - k;
            if (b < 0) b = 0;

            // Уменьшаем мощность антибиотика
            k = k - x;
            if (k < 0) k = 0;

            Console.WriteLine($"\nЧас {h}: Бактерий = {b}, Мощность антибиотика = {k}");

            // Возможность добавить бактерии или антибиотик
            Console.WriteLine("Хотите вмешаться в эксперимент?");
            Console.WriteLine("1 - Добавить бактерии");
            Console.WriteLine("2 - Добавить антибиотик");
            Console.WriteLine("0 - Продолжить без изменений");
            Console.Write("Ваш выбор: ");

            string choice = Console.ReadLine();

            if (choice == "1")
            {
                Console.Write("Сколько бактерий добавить? ");
                int addB = int.Parse(Console.ReadLine());
                b += addB;
                Console.WriteLine($"Добавлено {addB} бактерий. Теперь бактерий: {b}");
            }
            else if (choice == "2")
            {
                Console.Write("Сколько антибиотика добавить? ");
                int addX = int.Parse(Console.ReadLine());
                x += addX;
                k += addX * (11 - h); // Добавляем мощность пропорционально текущему часу
                Console.WriteLine($"Добавлено {addX} антибиотика. Теперь мощность: {k}");
            }

            if (b <= 0) break;
        }

        Console.WriteLine($"\nПроцесс завершен через {h} часов");
        Console.WriteLine($"Конечное количество бактерий: {b}");

        // Итоговый анализ
        Console.WriteLine("\nРезультат");
        if (b > 1000)
            Console.WriteLine("Бактерии выиграли");
        else if (b > 0)
            Console.WriteLine("Бактерии выжили, но под контролем");
        else
            Console.WriteLine("Бактерии уничтожены");
    }
}
