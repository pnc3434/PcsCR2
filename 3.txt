using System;

class Program
{
    static void Main()
    {
        Console.Write("Введите числитель M: ");
        int m = int.Parse(Console.ReadLine());

        Console.Write("Введите знаменатель N: ");
        int n = int.Parse(Console.ReadLine());

        int a = Math.Abs(m);// Находим НОД через вычитание
        int b = Math.Abs(n);

        while (a != b)
        {
            if (a > b)
                a = a - b;
            else
                b = b - a;
        }
        int d = a; 

        int p = m / d;// Сокращение дроби
        int q = n / d;

        if (q < 0)
        {
            p = -p;
            q = -q;
        }
        Console.WriteLine($"Несократимая дробь: {p}/{q}");
    }
}
