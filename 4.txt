using System;

class Program
{
    static void Main()
    {
        Console.WriteLine("Загадайте число от 0 до 63. Отвечайте на мои вопросы, и я его угадаю");
        Console.WriteLine("Отвечайте да или нет");

        int a = 0;    
        int b = 63;   
        int c = 0;    

        while (a <= b) // Бинарный поиск 
        {
            c = (a + b) / 2;  

            if (a == b)
            {
                c = a;
                break;
            }

            Console.Write($"Ваше число больше {c}? да или нет: ");
            string r = Console.ReadLine();

            if (r.ToLower() == "да")
            {
                a = c + 1;  
            }
            else if (r.ToLower() == "нет")
            {
                b = c;      
            }
            else
            {
                Console.WriteLine("Пожалуйста, введите да или нет");
            }
        }
        Console.WriteLine($"Ваше число: {c}");
        Console.WriteLine("Число угадано");
    }
}
