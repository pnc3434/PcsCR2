using System;

class Program
{
    static void Main()
    {
        int w1 = 300; // Параметры напитков
        int m1 = 0;   
        int p1 = 150; 
        int w2 = 30;  
        int m2 = 270; 
        int p2 = 170; 

        Console.Write("Введите кол-во мл Воды: ");// Ввод запасов
        int w = int.Parse(Console.ReadLine());

        Console.Write("Введите кол-во мл Молока: ");
        int m = int.Parse(Console.ReadLine());

        int c1 = 0; // Счетчики
        int c2 = 0; 
        int t = 0;  

        while (true) // Цикл обслуживаниа
        {
        
            bool canA = (w >= w1);
            bool canL = (w >= w2 && m >= m2);

            if (!canA && !canL)
            {
                Console.WriteLine("\nИнгредиенты закончились!");
                break;
            }

            // Вывод меню
            Console.WriteLine("\nМеню кофейни");
            if (canA) Console.WriteLine("1 - Американо (" + p1 + " руб)");
            if (canL) Console.WriteLine("2 - Латте (" + p2 + " руб)");
            Console.WriteLine("0 - Завершить работу");
            Console.Write("Выберите напиток: ");

            int v = int.Parse(Console.ReadLine());

            if (v == 0) break;

            if (v == 1 && canA)// Обрабатываем выбор
            {
                w -= w1;
                t += p1;
                c1++;
                Console.WriteLine("Американо готово");
            }
            else if (v == 2 && canL)
            {
                w -= w2;
                m -= m2;
                t += p2;
                c2++;
                Console.WriteLine("Латте готово");
            }
            else
            {
                Console.WriteLine("Недостаточно ингридиентов");
                continue;
            }

            // Показываем остатки
            Console.WriteLine("Остаток воды: " + w + " мл");
            Console.WriteLine("Остаток молока: " + m + " мл");
        }

        Console.WriteLine("\nОтчет смены");
        Console.WriteLine("Американо приготовлено: " + c1);
        Console.WriteLine("Латте приготовлено: " + c2);
        Console.WriteLine("Общая выручка: " + t + " руб");
        Console.WriteLine("Остаток воды: " + w + " мл");
        Console.WriteLine("Остаток молока: " + m + " мл");
    }
}
